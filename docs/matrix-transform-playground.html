import matplotlib.pyplot as plt
import numpy as np
from matplotlib.patches import Polygon
from shiny import App, reactive, render, ui

DEFAULT_MATRIX = (1.0, 0.6, -0.3, 1.0)
PRESETS = {
    "": "Preset...",
    "1,0,0,1": "Identity",
    "1,0.8,0,1": "Shear",
    "0,-1,1,0": "Rotate 90",
    "1,0,0,-1": "Reflect",
    "1,0,0,0": "Project",
    "1.2,0.7,-0.4,0.9": "Mixed",
}

ARROW_SHAPE = 0.58 * np.array(
    [
        [-1.6, -0.475],
        [0.225, -0.475],
        [0.225, -0.875],
        [1.475, 0],
        [0.225, 0.875],
        [0.225, 0.475],
        [-1.6, 0.475],
    ]
)
UNIT_SQUARE = np.array([[0, 0], [1, 0], [1, 1], [0, 1]])


def apply_matrix(matrix, points):
    return points @ matrix.T


def grid_segments(extent=2):
    segments = []
    for value in range(-extent, extent + 1):
        segments.append(
            (
                np.array([value, -extent], dtype=float),
                np.array([value, extent], dtype=float),
                "vertical",
            )
        )
        segments.append(
            (
                np.array([-extent, value], dtype=float),
                np.array([extent, value], dtype=float),
                "horizontal",
            )
        )
    return segments


def transformed_points_for_scale(matrix):
    points = []
    for start, end, _kind in grid_segments():
        points.extend([start, end])
        points.extend(apply_matrix(matrix, np.array([start, end])))
    points.extend(ARROW_SHAPE)
    points.extend(apply_matrix(matrix, ARROW_SHAPE))
    points.extend(UNIT_SQUARE)
    points.extend(apply_matrix(matrix, UNIT_SQUARE))
    points.extend([[1, 0], [0, 1]])
    points.extend(apply_matrix(matrix, np.array([[1, 0], [0, 1]])))
    return np.array(points)


def draw_grid(ax, matrix=None):
    vertical_light = "#cfd6e3" if matrix is None else "#2f6fdd"
    horizontal_light = "#cfd6e3" if matrix is None else "#e07a2f"
    vertical_strong = "#9aa4b2" if matrix is None else "#184fa3"
    horizontal_strong = "#9aa4b2" if matrix is None else "#b45309"

    for start, end, kind in grid_segments():
        points = np.array([start, end])
        if matrix is not None:
            points = apply_matrix(matrix, points)

        strong = abs(start[0]) < 1e-9 if kind == "vertical" else abs(start[1]) < 1e-9
        if strong and kind == "vertical":
            color = vertical_strong
        elif strong:
            color = horizontal_strong
        elif kind == "vertical":
            color = vertical_light
        else:
            color = horizontal_light

        ax.plot(points[:, 0], points[:, 1], color=color, linewidth=2.1 if strong else 1.1, alpha=0.9)


def draw_vector(ax, start, end, label, color):
    ax.annotate(
        "",
        xy=end,
        xytext=start,
        arrowprops={
            "arrowstyle": "-|>",
            "color": color,
            "linewidth": 3.0,
            "mutation_scale": 22,
        },
    )
    ax.text(
        end[0] + 0.08,
        end[1] + 0.08,
        label,
        color=color,
        fontsize=13,
        fontweight="bold",
    )


def draw_panel(ax, title, matrix=None):
    transform = (lambda points: points) if matrix is None else (lambda points: apply_matrix(matrix, points))
    draw_grid(ax, matrix)

    arrow = transform(ARROW_SHAPE)
    square = transform(UNIT_SQUARE)
    e1, e2 = transform(np.array([[1, 0], [0, 1]], dtype=float))

    ax.add_patch(
        Polygon(
            arrow,
            closed=True,
            facecolor="#d1d5db" if matrix is None else "#ddd6fe",
            edgecolor="#6b7280" if matrix is None else "#7c3aed",
            linewidth=2.6,
            alpha=0.65,
        )
    )
    ax.add_patch(
        Polygon(
            square,
            closed=True,
            facecolor="#99f6e4",
            edgecolor="#159a8a",
            linewidth=2.0,
            alpha=0.28,
        )
    )
    draw_vector(ax, np.array([0, 0]), e1, "e1" if matrix is None else "Ae1", "#159a8a")
    draw_vector(ax, np.array([0, 0]), e2, "e2" if matrix is None else "Ae2", "#b45309")
    ax.scatter([0], [0], color="#111827", s=32, zorder=5)
    ax.set_title(title, fontsize=15, fontweight="bold", loc="left")
    ax.set_aspect("equal")
    ax.set_xticks([])
    ax.set_yticks([])
    for spine in ax.spines.values():
        spine.set_color("#d6dde8")
        spine.set_linewidth(1.1)


def matrix_from_input(input):
    return np.array(
        [
            [input.a(), input.b()],
            [input.c(), input.d()],
        ],
        dtype=float,
    )

def server(input, output, session):
    @reactive.effect
    @reactive.event(input.reset)
    def _reset_matrix():
        ui.update_numeric("a", value=DEFAULT_MATRIX[0])
        ui.update_numeric("b", value=DEFAULT_MATRIX[1])
        ui.update_numeric("c", value=DEFAULT_MATRIX[2])
        ui.update_numeric("d", value=DEFAULT_MATRIX[3])

    @reactive.effect
    def _apply_preset():
        selected = input.preset()
        if not selected:
            return
        values = [float(value) for value in selected.split(",")]
        ui.update_numeric("a", value=values[0])
        ui.update_numeric("b", value=values[1])
        ui.update_numeric("c", value=values[2])
        ui.update_numeric("d", value=values[3])
        ui.update_select("preset", selected="")

    @output
    @render.ui
    def matrix_stats():
        matrix = matrix_from_input(input)
        det = float(np.linalg.det(matrix))
        orientation = "collapsed" if abs(det) < 1e-9 else "preserved" if det > 0 else "reversed"
        return ui.div(
            ui.div(ui.span("determinant"), ui.strong(f"{det:.3f}"), class_="matrix-stat"),
            ui.div(ui.span("area scale"), ui.strong(f"{abs(det):.3f}"), class_="matrix-stat"),
            ui.div(ui.span("orientation"), ui.strong(orientation), class_="matrix-stat"),
            class_="matrix-stats",
        )

    @output
    @render.plot(alt="Side-by-side plots of the original grid and arrow shape and their transformed image under a two by two matrix.")
    def matrix_plot():
        matrix = matrix_from_input(input)
        fig, axes = plt.subplots(1, 2, figsize=(14, 7))

        draw_panel(axes[0], "Original")
        draw_panel(axes[1], "Transformed by A", matrix)

        all_points = transformed_points_for_scale(matrix)
        limit = max(2.2, float(np.max(np.abs(all_points))) * 1.15)
        for ax in axes:
            ax.set_xlim(-limit, limit)
            ax.set_ylim(-limit, limit)

        fig.patch.set_facecolor("#ffffff")
        fig.tight_layout()
        return fig


app = App(app_ui, server)
